package belote.ex.controller;

import belote.ex.business.StatisticsServiceInt;
import belote.ex.domain.GetAllGameStatistics;
import belote.ex.persistance.entity.WonGames;
import jakarta.annotation.security.RolesAllowed;
import lombok.AllArgsConstructor;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RestController;

import java.util.List;


@RestController
@RequestMapping("/stats")
@AllArgsConstructor
@RolesAllowed({"ADMIN"})
public class StatisticsController {

    StatisticsServiceInt statisticsService;

    @GetMapping
    public GetAllGameStatistics getAllGames() {
        return GetAllGameStatistics.builder().games(statisticsService.getAllGames()).build();
    }

    @GetMapping("{id}")
    public GetAllGameStatistics getGamesWonByUserID(@PathVariable int id) {
        return GetAllGameStatistics.builder().games(statisticsService.getWonGamesByUserID(id)).build();
    }

    @GetMapping("username/{username}")
    public GetAllGameStatistics getGamesWonByUsername(@PathVariable String username) {
        return GetAllGameStatistics.builder().games(statisticsService.getWonGamesByUsername(username)).build();
    }

    @GetMapping("leaderBoard")
    public List<WonGames> getGamesWonByUsername() {
        return statisticsService.getWonGamesForAllUsers();
    }

}
