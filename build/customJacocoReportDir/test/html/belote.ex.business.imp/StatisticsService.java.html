package belote.ex.business.imp;

import belote.ex.business.StatisticsServiceInt;
import belote.ex.persistance.GameScoresRepository;
import belote.ex.persistance.entity.GameScoreEntity;
import belote.ex.persistance.entity.WonGames;
import belote.ex.persistance.repo.GameRepository;
import lombok.AllArgsConstructor;
import org.springframework.messaging.simp.SimpMessagingTemplate;
import org.springframework.scheduling.annotation.EnableScheduling;
import org.springframework.scheduling.annotation.Scheduled;
import org.springframework.stereotype.Service;

import java.util.List;
import java.util.Random;

@Service
@AllArgsConstructor
@EnableScheduling
public class StatisticsService implements StatisticsServiceInt {

    private final SimpMessagingTemplate messagingTemplate;

    private GameScoresRepository gameScoresRepositoryrepository;
    private GameRepository gameRepository;

    @Override
    public List<GameScoreEntity> getAllGames() {
        return gameScoresRepositoryrepository.findAll();
    }

    @Override
    public List<GameScoreEntity> getWonGamesByUserID(int id) {
        return gameScoresRepositoryrepository.findGameScoreEntitiesWonByUserID(id);
    }

    @Override
    public List<GameScoreEntity> getWonGamesByUsername(String username) {
        return  gameScoresRepositoryrepository.findGameScoreEntitiesWonByUsername(username);
    }

    @Override
    public List<WonGames> getWonGamesForAllUsers() {
        return gameScoresRepositoryrepository.findGamesWonForAllUsers();
    }

    @Scheduled(fixedRate = 5000)
    public void performTask() {
        //Idea for live statistics
        //Added random for presentation purposes
        Random rand = new Random();
        int max=10;
        int min=1;
        String to = "/stats/live" ;
        messagingTemplate.convertAndSend(to, getAllCurrentGames() + rand.nextInt(max - min + 1) + min);
    }

    public int getNumberOfAllGames()
    {
        return  gameScoresRepositoryrepository.findAll().size();
    }

    public int getAllCurrentGames(){
        return gameRepository.getAllGames().size();
    }
}
