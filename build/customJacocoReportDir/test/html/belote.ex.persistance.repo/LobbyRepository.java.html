package belote.ex.persistance.repo;


import belote.ex.persistance.LobbyRepositoryInt;
import belote.ex.persistance.entity.LobbyEntity;
import belote.ex.persistance.entity.UserEntity;
import org.springframework.stereotype.Repository;

import java.util.ArrayList;
import java.util.List;

@Repository
public class LobbyRepository implements LobbyRepositoryInt {
    private static int id = 1;
    private final List<LobbyEntity> lobbies = new ArrayList<>();
    public int createLobby(UserEntity user){
        ArrayList<UserEntity> users = new ArrayList<>();
        users.add(user);
        lobbies.add(LobbyEntity.builder().id(id).players(users).build());
        nextID();
        return id-1;
    }

    public static void nextID() {
        id++;
    }
    public void joinLobby(UserEntity user, int lobbyID){

        getLobby(lobbyID).getPlayers().add(user);
    }

    public LobbyEntity getLobby(int id){
      return   lobbies.stream().filter(x -> x.getId() == id).findFirst().orElse(null);
    }

    public void deleteLobby(int id){
        LobbyEntity lobby = getLobby(id);
        lobbies.remove(lobby);
    }
}
